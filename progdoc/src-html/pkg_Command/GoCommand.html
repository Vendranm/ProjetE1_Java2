011public class GoCommand extends Command
012{
013    // variables d'instance - remplacez l'exemple qui suit par le vôtre
014
015    /**
016     * Constructeur d'objets de classe GoCommand
017     */
018    public GoCommand()
019    {
020
021    }
022    
023    /**
024     * Override method that execute the command go
025     */
026    @Override
027    public void execute(final Player pPlayer) 
028    {
029        if(!hasSecondWord()) 
030        {
031            // if there is no second word, we don't know where to go...
032            GameEngine.aGui.println("Go where?");
033            return;
034        }
035
036        String vDirection = getSecondWord();
037
038        // Try to leave current room.
039        Room vPreviousRoom= pPlayer.getCurrentRoom();
040        pPlayer.getBackRoom().push(vPreviousRoom);
041        Room vNextRoom = pPlayer.getCurrentRoom().getExit(vDirection);
042
043        if (vNextRoom == null)
044            GameEngine.aGui.println("There is no door!");
045        
046        else if(! pPlayer.getInventaire().hasItem("key") && pPlayer.getCurrentRoom().getDescription().equals(" in a special room which leads nowhere..") && vDirection.equals("West"))
047        {
048            GameEngine.aGui.println("You can't go this way. The door is closed. You need the key.");
049            return;
050        }
051        else 
052        {
053            pPlayer.setCurrentRoom(vNextRoom);
054            GameEngine.printLocationInfo();
055            
056            if(pPlayer.getCurrentRoom().getImageName() != null)
057            {
058                GameEngine.aGui.showImage(pPlayer.getCurrentRoom().getImageName());
059            }
060            GameEngine.hurt();
061        }
062    }//goRoom
063}
