001package pkg_Command;
002
003import java.util.HashMap;
004/**
005 * This class is the main class of the "World of Zuul" application. 
006 * "World of Zuul" is a very simple, text based adventure game.
007 * 
008 * This class holds an enumeration of all command words known to the game.
009 * It is used to recognise commands as they are typed in.
010 *
011 * @author  Michael Kolling and David J. Barnes
012 * @version 1.0 (February 2002) DBMOD: 04/04/2008
013 */
014
015public class CommandWords
016{
017    // a constant array that holds all valid command words
018    private HashMap<String, CommandWord> aValidCommands;
019
020    /**
021     * Constructor - initialise the command words.
022     */
023    public CommandWords()
024    {
025        aValidCommands = new HashMap<String, CommandWord>();
026        for(CommandWord vCommand : CommandWord.values()) 
027        {
028            if(vCommand != CommandWord.UNKNOWN) 
029            {
030                aValidCommands.put(vCommand.toString(), vCommand);
031            }
032        }
033    }
034    
035    /**
036     * Find the CommandWord associated with a command word.
037     * @param commandWord The word to look up.
038     * @return The CommandWord correspondng to commandWord, or UNKNOWN
039     *         if it is not a valid command word.
040     */
041    public CommandWord getCommandWord(final String pCommandWord)
042    {
043        CommandWord vCommand = aValidCommands.get(pCommandWord);
044        if(vCommand != null) 
045        {
046            return vCommand;
047        }
048        else 
049        {
050            return CommandWord.UNKNOWN;
051        }
052    }
053    
054    /**
055     * Check whether a given String is a valid command word. 
056     * @return true if it is, false if it isn't.
057     */
058    public boolean isCommand(final String aString)
059    {
060        return aValidCommands.containsKey(aString);
061    }
062
063    /**
064     * returns a String of all valid commands.
065     */
066    public String getCommandList() 
067    {
068        StringBuilder vCommands = new StringBuilder();
069        for(String command : aValidCommands.keySet()) 
070        {
071            vCommands.append(command + " ");
072        }
073        //         for(int i = 0; i < aValidCommands.size(); i++) {
074        //             commands.append( aValidCommands[i] + "  " );
075        //         }
076        return vCommands.toString();
077    }
078    
079    /**
080     * Return a Command from a commandWord
081     */
082    public Command getCommand(final String pCommandWord)
083    {
084        return this.getCommandWord(pCommandWord).getCommand();
085    }
086}
