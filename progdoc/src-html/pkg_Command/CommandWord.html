010public enum CommandWord
011{
012    GO("go", new GoCommand()),
013    QUIT("quit", new QuitCommand()),
014    HELP("help", new HelpCommand()), 
015    LOOK("look", new LookCommand()), 
016    EAT("eat", new EatCommand()),
017    GRAB("grab", new GrabCommand()),
018    DROP("drop", new DropCommand()),
019    BACK("back", new BackCommand()),
020    TEST("test", new TestCommand()),
021    UNKNOWN("?", null),
022    CHARGE("charge", new ChargeCommand()),
023    TELEPORT("teleport", new TeleportCommand()),
024    ALEA("alea", new AleaCommand());
025    
026    
027    private String aCommandString;
028    private Command aCommand;
029    /**
030     * Initialise with the corresponding command word.
031     * @param commandWord The command string.
032     */
033    CommandWord(final String pCommandString, final Command pC)
034    {
035        this.aCommandString = pCommandString;
036        this.aCommand = pC;
037    }
038    
039    /**
040     * @return The command word as a string.
041     */
042    public String toString()
043    {
044        return this.aCommandString;
045    }
046    
047    /**
048     * Returns the command aCommand
049     */
050    public Command getCommand()
051    {
052        return this.aCommand;
053    }
